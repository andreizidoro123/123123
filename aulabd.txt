CREATE TABLE Forncedor (
  ID INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  FORNECEDOR VARCHAR(100) NULL,
  PRIMARY KEY(ID)
);

CREATE TABLE Forncedor_has_Produto (
  Forncedor_ID INTEGER UNSIGNED NOT NULL,
  Produto_ID INTEGER UNSIGNED NOT NULL,
  PRIMARY KEY(Forncedor_ID, Produto_ID),
  INDEX Forncedor_has_Produto_FKIndex1(Forncedor_ID),
  INDEX Forncedor_has_Produto_FKIndex2(Produto_ID)
);
